// 1. Instale e importe
import { MockAPI } from '@mockapi/js'

// 2. Conecte com seu token
const mock = new MockAPI('mapi_seu_token_aqui')

// 3. Defina seus mocks
await mock.intercept('GET /users', {
  status: 200,
  data: mock.faker('user', 10)   // gera 10 users fake
})

await mock.intercept('POST /login', {
  status: 200,
  body: { token: 'jwt-simulado-123', expiresIn: 3600 }
})

// 4. Use normalmente — sem mudar seu código de produção
const res = await fetch(mock.url('/users'))
const users = await res.json()  // → [{id, name, email, ...}]

// 5. Limpe após os testes
await mock.reset()
import { MockAPI, MockConfig } from '@mockapi/js'

const mock = new MockAPI('mapi_seu_token')

// Tipagem completa com generics
interface User {
  id: number; name: string; email: string
}

await mock.intercept<User[]>('GET /users', {
  data: mock.faker<User>('user', 10)
})

// CRUD automático com tipagem
const users = mock.crud<User>('/api/users')
await users.seed([
  { id: 1, name: 'Ana', email: 'ana@teste.com' }
])
